93 chig::Result chig::Context::loadModule(const gsl::cstring_span<> name, ChigModule** toFill) {
94  Result res;
95 
96  // check for built-in modules
97  if (name == "lang") {
98  auto mod = std::make_unique<LangModule>(*this);
99  if (toFill != nullptr) { *toFill = mod.get(); }
100  addModule(std::move(mod));
101  return {};
102  }
103  if (name == "c") {
104  auto mod = std::make_unique<CModule>(*this);
105  if (toFill != nullptr) { *toFill = mod.get(); }
106  addModule(std::move(mod)); // we don't care if it's actually added
107  return {};
108  }
109 
110  if (workspacePath().empty()) {
111  res.addEntry("EUKN", "Cannot load module without a workspace path",
112  {{"Requested Module"}, gsl::to_string(name)});
113  return res;
114  }
115 
116  // find it in the workspace
117  fs::path fullPath = workspacePath() / "src" / (gsl::to_string(name) + ".chigmod");
118 
119  if (!fs::is_regular_file(fullPath)) {
120  res.addEntry(
121  "EUKN", "Failed to find module",
122  {{"Module Name", gsl::to_string(name)}, {"Workspace Path", workspacePath().string()}});
123  return res;
124  }
125 
126  // load the JSON
127  nlohmann::json readJson = {};
128  {
129  fs::ifstream inFile{fullPath};
130 
131  inFile >> readJson;
132  }
133 
134  JsonModule* toFillJson = nullptr;
135  res += addModuleFromJson(name, readJson, &toFillJson);
136  if (toFill != nullptr) { *toFill = toFillJson; }
137 
138  return res;
139 }

141 Result Context::addModuleFromJson(gsl::cstring_span<> fullName, const nlohmann::json& json,
142  JsonModule** toFill) {
143  Result res;
144 
145  // make sure it's not already added
146  {
147  auto mod = moduleByFullName(fullName);
148  if (mod != nullptr) {
149  if (toFill != nullptr) {
150  auto casted = dynamic_cast<JsonModule*>(mod);
151  if (casted != nullptr) { *toFill = casted; }
152  }
153  return {};
154  }
155  }
156 
157  // Create the module
158  JsonModule* cPtr = nullptr;
159  {
160  // parse module
161  auto jmod = std::make_unique<JsonModule>(*this, gsl::to_string(fullName), json, &res);
162  if (!res) { return res; }
163  if (toFill != nullptr) { *toFill = jmod.get(); }
164 
165  cPtr = jmod.get();
166  bool added = addModule(std::move(jmod));
167  Expects(added); // it really should be added
168  }
169 
170  // load graphs
171  res += cPtr->loadGraphs();
172 
173  return res;

248 Result Context::compileModule(gsl::cstring_span<> fullName, std::unique_ptr<llvm::Module>* toFill) {
249  Expects(toFill != nullptr);
250 
251  Result res;
252 
253  auto chigmod = moduleByFullName(fullName);
254 
255  if (chigmod == nullptr) {
256  res.addEntry("E36", "Could not find module", {{"module", gsl::to_string(fullName)}});
257  return res;
258  }
259 
260  auto llmod = std::make_unique<llvm::Module>(gsl::to_string(fullName), llvmContext());
261 
262  // generate dependencies
263  for (const auto& depName : chigmod->dependencies()) {
264  std::unique_ptr<llvm::Module> compiledDep;
265  res += compileModule(depName, &compiledDep); // TODO: detect circular dependencies
266 
267  if (!res) { return res; }
268 
269 
270  // link it in
271 #if LLVM_VERSION_MAJOR <= 3 && LLVM_VERSION_MINOR <= 7
272  llvm::Linker::LinkModules(llmod.get(), compiledDep.get());
273 #else
274  llvm::Linker::linkModules(*llmod, std::move(compiledDep));
275 #endif
276  }
277 
278  res += chigmod->generateModule(*llmod);
279 
280  // set debug info version if it doesn't already have it
281  if (llmod->getModuleFlag("Debug Info Version") == nullptr) {
282  llmod->addModuleFlag(llvm::Module::Warning, "Debug Info Version",
283  llvm::DEBUG_METADATA_VERSION);
284  }
285 
286  // verify the created module
287  if (res) {
288  std::string err;
289  llvm::raw_string_ostream os(err);
290  if (llvm::verifyModule(*llmod, &os)) {
291  std::string moduleStr;
292  {
293  llvm::raw_string_ostream printerStr{moduleStr};
294  llmod->print(printerStr, nullptr);
295  }
296  res.addEntry(
297  "EINT", "Internal compiler error: Invalid module created",
298  {{"Error", err}, {"Full Name", gsl::to_string(fullName)}, {"Module", moduleStr}});
299  }
300  }
301 
302  *toFill = std::move(llmod);
303 
304  return res;
305 }
