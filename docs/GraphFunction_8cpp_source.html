32 GraphFunction::GraphFunction(JsonModule& module, const nlohmann::json& data, Result& res) {
33  if (!data.is_object()) {
34  res.addEntry("E1", "Graph json isn't a JSON object", {});
35  return;
36  }
37  // make sure it has a type element
38  if (data.find("type") == data.end()) {
39  res.addEntry("E2", R"(JSON in graph doesn't have a "type" element)", {});
40  return;
41  }
42  if (data["type"] != "function") {
43  res.addEntry("E3", "JSON in graph doesn't have a function type", {});
44  return;
45  }
46  // make sure there is a name
47  if (data.find("name") == data.end()) {
48  res.addEntry("E4", "JSON in graph doesn't have a name parameter", {});
49  return;
50  }
51  std::string name = data["name"];
52 
53  if (data.find("data_inputs") == data.end() || !data["data_inputs"].is_array()) {
54  res.addEntry("E43", "JSON in graph doesn't have an data_inputs array", {});
55  return;
56  }
57 
58  std::vector<std::pair<DataType, std::string>> datainputs;
59  for (auto param : data["data_inputs"]) {
60  for (auto iter = param.begin(); iter != param.end(); ++iter) {
61  std::string qualifiedType = iter.value();
62  std::string docString = iter.key();
63 
64  std::string moduleName, name;
65  std::tie(moduleName, name) = parseColonPair(qualifiedType);
66 
67  DataType ty;
68  res += module.context().typeFromModule(moduleName, name, &ty);
69 
70  if (!res) { return; }
71 
72  datainputs.emplace_back(ty, docString);
73  }
74  }
75 
76  if (data.find("data_outputs") == data.end() || !data["data_outputs"].is_array()) {
77  res.addEntry("E44", "JSON in graph doesn't have an data_outputs array", {});
78  return;
79  }
80 
81  std::vector<std::pair<DataType, std::string>> dataoutputs;
82  for (auto param : data["data_outputs"]) {
83  for (auto iter = param.begin(); iter != param.end(); ++iter) {
84  std::string qualifiedType = iter.value();
85  std::string docString = iter.key();
86 
87  std::string moduleName, name;
88  std::tie(moduleName, name) = parseColonPair(qualifiedType);
89 
90  DataType ty;
91  res += module.context().typeFromModule(moduleName, name, &ty);
92 
93  if (!res) { return; }
94 
95  dataoutputs.emplace_back(ty, docString);
96  }
97  }
98 
99  // get exec I/O
100  if (data.find("exec_inputs") == data.end() || !data["exec_inputs"].is_array()) {
101  res.addEntry("EUKN", "JSON in graph doesn't have an exec_inputs array", {});
102  return;
103  }
104 
105  std::vector<std::string> execinputs;
106  for (const auto& param : data["exec_inputs"]) {
107  std::string name = param;
108 
109  execinputs.emplace_back(name);
110  }
111 
112  if (data.find("exec_outputs") == data.end() || !data["exec_outputs"].is_array()) {
113  res.addEntry("EUKN", "JSON in graph doesn't have an data_outputs array", {});
114  return;
115  }
116 
117  std::vector<std::string> execoutputs;
118  for (const auto& param : data["exec_outputs"]) {
119  std::string name = param;
120 
121  execoutputs.emplace_back(name);
122  }
123 
124  // construct it
125  mModule = &module;
126  mContext = &module.context();
127  mName = name;
128  mDataInputs = std::move(datainputs);
129  mDataOutputs = std::move(dataoutputs);
130  mExecInputs = std::move(execinputs);
131  mExecOutputs = std::move(execoutputs);
132 
133  mSource = data;
134 }

175 NodeInstance* GraphFunction::entryNode() const noexcept {
176  auto matching = graph().nodesWithType("lang", "entry");
177 
178  if (matching.size() == 1) {
179  auto span = matching[0]->type().dataOutputs();
180  // make sure it has the same signature as the method
181  if (!std::equal(dataInputs().begin(), dataInputs().end(), span.begin(), span.end())) {
182  return nullptr;
183  }
184  // make sure it has the same exec names and size
185  if (!std::equal(execInputs().begin(), execInputs().end(),
186  matching[0]->type().execOutputs().begin())) {
187  return nullptr;
188  }
189  return matching[0];
190  }
191  return nullptr;
192 }

205 Result GraphFunction::removeNode(NodeInstance* nodeToRemove) {
206  Result res;
207 
208  // disconnect it's connections
209 
210  // disconnect input exec
211  for (const auto& execSlot : nodeToRemove->inputExecConnections) {
212  for (const auto& pair : execSlot) {
213  if (pair.first != nullptr) { res += disconnectExec(*pair.first, pair.second); }
214  }
215  }
216  // disconnect output exec
217  auto ID = 0ull;
218  for (const auto& pair : nodeToRemove->outputExecConnections) {
219  if (pair.first != nullptr) { res += disconnectExec(*nodeToRemove, ID); }
220  ++ID;
221  }
222 
223  // disconnect input data
224  for (const auto& pair : nodeToRemove->inputDataConnections) {
225  if (pair.first != nullptr) {
226  res += disconnectData(*pair.first, pair.second, *nodeToRemove);
227  }
228  }
229 
230  // disconnect output data
231  ID = 0ull;
232  for (const auto& dataSlot : nodeToRemove->outputDataConnections) {
233  for (const auto& pair : dataSlot) {
234  if (pair.first != nullptr) { disconnectData(*nodeToRemove, ID, *pair.first); }
235  }
236  ++ID;
237  }
238  // then delete the node
239  graph().nodes().erase(nodeToRemove->id());
240 
241  return res;
242 }

355 
356  // this is for which exec input
