30 GraphFunction::GraphFunction(JsonModule& module, const nlohmann::json& data, Result& res)
31 {
32  if (!data.is_object()) {
33  res.addEntry("E1", "Graph json isn't a JSON object", {});
34  return;
35  }
36  // make sure it has a type element
37  if (data.find("type") == data.end()) {
38  res.addEntry("E2", R"(JSON in graph doesn't have a "type" element)", {});
39  return;
40  }
41  if (data["type"] != "function") {
42  res.addEntry("E3", "JSON in graph doesn't have a function type", {});
43  return;
44  }
45  // make sure there is a name
46  if (data.find("name") == data.end()) {
47  res.addEntry("E4", "JSON in graph doesn't have a name parameter", {});
48  return;
49  }
50  std::string name = data["name"];
51 
52  if (data.find("data_inputs") == data.end() || !data["data_inputs"].is_array()) {
53  res.addEntry("E43", "JSON in graph doesn't have an data_inputs array", {});
54  return;
55  }
56 
57  std::vector<std::pair<DataType, std::string>> datainputs;
58  for (auto param : data["data_inputs"]) {
59  for (auto iter = param.begin(); iter != param.end(); ++iter) {
60  std::string qualifiedType = iter.value();
61  std::string docString = iter.key();
62 
63  std::string moduleName, name;
64  std::tie(moduleName, name) = parseColonPair(qualifiedType);
65 
66  DataType ty;
67  res += module.context().typeFromModule(moduleName, name, &ty);
68 
69  if (!res) {
70  return;
71  }
72 
73  datainputs.emplace_back(ty, docString);
74  }
75  }
76 
77  if (data.find("data_outputs") == data.end() || !data["data_outputs"].is_array()) {
78  res.addEntry("E44", "JSON in graph doesn't have an data_outputs array", {});
79  return;
80  }

83  for (auto param : data["data_outputs"]) {
84  for (auto iter = param.begin(); iter != param.end(); ++iter) {
85  std::string qualifiedType = iter.value();

87

187 // Codegens a single input to a node
188 void codegenHelper(NodeInstance* node, unsigned execInputID, llvm::BasicBlock* block,
189  llvm::BasicBlock* allocblock, llvm::Module* mod, llvm::Function* f,
190  std::unordered_map<NodeInstance*, cache>& nodeCache, Result& res)
191 {
192  llvm::IRBuilder<> builder(block);
193 
194  // get inputs and outputs
195  std::vector<llvm::Value*> io;
196 
197  // add inputs
198  {
199  size_t inputID = 0;
200  for (auto& param : node->inputDataConnections) {
201  // make sure everything is A-OK
202  if (param.first == nullptr) {
203  res.addEntry("EUKN", "No data input to node",
204  {{"nodeid", node->id()}, {"input ID", inputID},
205  {"nodetype", node->type().qualifiedName()}});
206 
207  return;
208  }
209 
210  auto cacheiter = nodeCache.find(param.first);
211 
212  if (cacheiter == nodeCache.end()) {
213  res.addEntry("EUKN", "Failed to find in cache", {{"nodeid", param.first->id()}});
214 
215  return;
216  }
217 
218  auto& cacheObject = cacheiter->second;
219  if (param.second >= cacheObject.outputs.size()) {
220  res.addEntry("EUKN", "No data input to node",
221  {{"nodeid", node->id()}, {"input ID", inputID},
222  {"nodetype", node->type().qualifiedName()}});
223 
224  return;
225  }
226 
227  // get pointers to the objects
228  auto value = cacheObject.outputs[param.second];
229  // dereference
230  io.push_back(builder.CreateLoad(
231  value, node->type().dataInputs()[inputID].second)); // TODO: pass ptr to value
232 
233  // make sure it's the right type
234  if (io[io.size() - 1]->getType() !=
235  node->type().dataInputs()[inputID].first.llvmType()) {
236  res.addEntry("EINT", "Internal codegen error: unexpected type in cache.",
237  {{"Expected LLVM type",
238  stringifyLLVMType(node->type().dataInputs()[inputID].first.llvmType())},
239  {"Found type", stringifyLLVMType(io[io.size() - 1]->getType())},
240  {"Node ID", node->id()}, {"Input ID", inputID}});
241  }
242 
243  ++inputID;
244  }
245  }
246 
247  // get outputs
248  {
249  llvm::IRBuilder<> allocBuilder(allocblock);
250 
251  // create outputs
252  auto& outputCache = nodeCache[node].outputs;
253 
254  for (auto& output : node->type().dataOutputs()) {
255  // TODO: research address spaces
256  llvm::AllocaInst* alloc =
257  allocBuilder.CreateAlloca(output.first.llvmType(), nullptr, output.second);
258  alloc->setName(output.second);
259  outputCache.push_back(alloc);
260  io.push_back(alloc);
261 
262  // make sure the type is right
263  if (llvm::PointerType::get(output.first.llvmType(), 0) != alloc->getType()) {
264  res.addEntry("EINT",
265  "Internal codegen error: unexpected type returned from alloca.",
266  {{"Expected LLVM type",
267  stringifyLLVMType(llvm::PointerType::get(output.first.llvmType(), 0))},
268  {"Yielded type", stringifyLLVMType(alloc->getType())},
269  {"Node ID", node->id()}});
270  }
271  }
272  }
273 
274  // add this block to the cache
275  if (node->type().qualifiedName() != "lang:entry") {
276  nodeCache[node].inputBlock.resize(node->inputExecConnections.size(), nullptr);
277  nodeCache[node].inputBlock[execInputID] = block;
278  }
279 
280  // create output blocks
281  std::vector<llvm::BasicBlock*> outputBlocks;
282  std::vector<llvm::BasicBlock*> unusedBlocks;
283  boost::dynamic_bitset<>
284  needsCodegen; // this holds if each block needs to recurse of it it's already in the cache
285  for (auto idx = 0ull; idx < node->outputExecConnections.size(); ++idx) {
286  // see if we can find it in the cache
287  NodeInstance* remoteNode;
288  size_t remotePortIdx;
289  std::tie(remoteNode, remotePortIdx) = node->outputExecConnections[idx];
290 
291  auto remoteCache = nodeCache[remoteNode]; // this will construct it if it doesn't exist
292  if (remoteCache.inputBlock.size() > remotePortIdx &&
293  remoteCache.inputBlock[remotePortIdx] != nullptr) {
294  // this means it's already in the cache
295  outputBlocks.push_back(remoteCache.inputBlock[remotePortIdx]);
296  needsCodegen.push_back(false);
297  } else {
298  auto outBlock =
299  llvm::BasicBlock::Create(f->getContext(), node->type().execOutputs()[idx], f);
300  outputBlocks.push_back(outBlock);
301  needsCodegen.push_back(true); // these need codegen
302  if (node->outputExecConnections[idx].first != nullptr) {
303  outBlock->setName("node_" + node->outputExecConnections[idx].first->id());
304  } else {
305  unusedBlocks.push_back(outBlock);
306  }
307  }
308  }
309 
310  // codegen
311  res += node->type().codegen(execInputID, mod, f, io, block, outputBlocks);
312  if (!res) {
313  return;
314  }
315 
316  // TODO: sequence nodes
317  for (auto& bb : unusedBlocks) {
318  llvm::IRBuilder<> builder(bb);
319 
320  builder.CreateRet(
321  llvm::ConstantInt::get(llvm::IntegerType::getInt32Ty(mod->getContext()), 0, true));
322  }
323 
324  // recurse!
325  for (auto idx = 0ull; idx < node->outputExecConnections.size(); ++idx) {
326  auto& output = node->outputExecConnections[idx];
327  if (output.first != nullptr && needsCodegen[idx]) {
328  codegenHelper(
329  output.first, output.second, outputBlocks[idx], allocblock, mod, f, nodeCache, res);
330  }
331  }
332 }

334 Result GraphFunction::compile(llvm::Module* mod, llvm::Function** ret_func) const
335 {
336  Result res;
337 
338  auto entry = entryNode();
339  if (entry == nullptr) {
340  res.addEntry("EUKN", "No entry node", {});
341  return res;
342  }
343 
344  // make sure that the entry node has the functiontype
345  if (!std::equal(
346  dataInputs().begin(), dataInputs().end(), entry->type().dataOutputs().begin())) {
347  nlohmann::json inFunc = nlohmann::json::array();
348  for (auto& in : dataInputs()) {
349  inFunc.push_back({{in.second, in.first.qualifiedName()}});
350  }
351 
352  nlohmann::json inEntry = nlohmann::json::array();
353  for (auto& in :
354  entry->type().dataOutputs()) { // outputs to entry are inputs to the function
355  inEntry.push_back({{in.second, in.first.qualifiedName()}});
356  }
357 
358  res.addEntry("EUKN", "Inputs to function doesn't match function inputs",
359  {{"Function Inputs", inFunc}, {"Entry Inputs", inEntry}});
360  return res;
361  }
362 
363  // make sure that the entry node has the functiontype
364  if (!std::equal(dataOutputs().begin(), dataOutputs().end(), dataOutputs().begin())) {
365  nlohmann::json outFunc = nlohmann::json::array();
366  for (auto& out : dataOutputs()) {
367  outFunc.push_back({{out.second, out.first.qualifiedName()}});
368  }
369 
370  nlohmann::json outEntry = nlohmann::json::array();
371  for (auto& out : dataOutputs()) {
372  // inputs to the exit are outputs to the function
373  outEntry.push_back({{out.second, out.first.qualifiedName()}});
374  }
375 
376  res.addEntry("EUKN", "Outputs to function doesn't match function exit",
377  {{"Function Outputs", outFunc}, {"Entry Outputs", outEntry}});
378  return res;
379  }
380 
381  llvm::Function* f = llvm::cast<llvm::Function>(mod->getOrInsertFunction(
382  mangleFunctionName(module().fullName(), name()), functionType())); // TODO: name mangling
383  llvm::BasicBlock* allocblock = llvm::BasicBlock::Create(mod->getContext(), "alloc", f);
384  llvm::BasicBlock* block = llvm::BasicBlock::Create(mod->getContext(), name() + "_entry", f);
385  auto blockcpy = block;
386 
387  // follow "linked list"
388  unsigned execInputID =
389  entry->outputExecConnections[0].second; // the exec connection to codegen from
390 
391  std::unordered_map<NodeInstance*, cache> nodeCache;
392 
393  // set argument names
394  auto idx = 0ull;
395  for (auto& arg : f->getArgumentList()) {
396  // the first one is the input exec ID
397  if (idx == 0) {
398  arg.setName("inputexec_id");
399  ++idx;
400  continue;
401  }
402  // all the - 1's is becaues the first is the inputexec_id
403  if (idx - 1 < entry->type().dataOutputs().size()) {
404  // dataOutputs to entry are inputs to the function
405  arg.setName(entry->type()
406  .dataOutputs()[idx - 1]
407  .second); // it starts with inputs, which are outputs to entry
408  } else {
409  arg.setName(dataOutputs()[idx - 1 - entry->type().dataOutputs().size()]
410  .second); // then the outputs, which are inputs to exit
411  }
412  ++idx;
413  }
414 
415  codegenHelper(entry, execInputID, block, allocblock, mod, f, nodeCache, res);
416 
417  llvm::IRBuilder<> allocbuilder(allocblock);
418  allocbuilder.CreateBr(blockcpy);
419 
420  *ret_func = f;
421  return res;
422 }

424 NodeInstance* GraphFunction::entryNode() const noexcept
425 {
426  auto matching = graph().nodesWithType("lang", "entry");
427 
428  if (matching.size() == 1) {
429  auto span = matching[0]->type().dataOutputs();
430  // make sure it has the same signature as the method
431  if (!std::equal(dataInputs().begin(), dataInputs().end(), span.begin(), span.end())) {
432  return nullptr;
433  }
434  // make sure it has the same exec names and size
435  if (!std::equal(execInputs().begin(), execInputs().end(),
436  matching[0]->type().execOutputs().begin())) {
437  return nullptr;
438  }
439  return matching[0];
440  }
441  return nullptr;
442 }

457 Result GraphFunction::removeNode(NodeInstance* nodeToRemove)
458 {
459  Result res;
460 
461  // disconnect it's connections
462 
463  // disconnect input exec
464  for (const auto& execSlot : nodeToRemove->inputExecConnections) {
465  for (const auto& pair : execSlot) {
466  if (pair.first != nullptr) {
467  res += disconnectExec(*pair.first, pair.second);
468  }
469  }
470  }
471  // disconnect output exec
472  auto ID = 0ull;
473  for (const auto& pair : nodeToRemove->outputExecConnections) {
474  if (pair.first != nullptr) {
475  res += disconnectExec(*nodeToRemove, ID);
476  }
477  ++ID;
478  }
479 
480  // disconnect input data
481  for (const auto& pair : nodeToRemove->inputDataConnections) {
482  if (pair.first != nullptr) {
483  res += disconnectData(*pair.first, pair.second, *nodeToRemove);
484  }
485  }
486 
487  // disconnect output data
488  ID = 0ull;
489  for (const auto& dataSlot : nodeToRemove->outputDataConnections) {
490  for (const auto& pair : dataSlot) {
491  if (pair.first != nullptr) {
492  disconnectData(*nodeToRemove, ID, *pair.first);
493  }
494  }
495  ++ID;
496  }
497  // then delete the node
498  graph().nodes().erase(nodeToRemove->id());
499 
500  return res;
501 }

577 Result GraphFunction::validateGraph() const
578 {
579  Result res;
580  // make sure all connections connect back
581 
582  for (const auto& node : graph().nodes()) {
583  // go through input data
584  auto id = 0ull;
585  for (const auto& conn : node.second->inputDataConnections) {
586  // make sure it connects back
587  bool connectsBack = false;
588  for (const auto& remoteConn : conn.first->outputDataConnections[conn.second]) {
589  if (remoteConn.first == node.second.get() && remoteConn.second == id) {
590  connectsBack = true;
591  break;
592  }
593  }
594  if (!connectsBack) {
595  res.addEntry("EUKN", "Data connection doesn't connect back",
596  {{"Left Node", conn.first->id()}, {"Right Node", node.second->id()},
597  {"Right input ID", id}});
598  }
599  ++id;
600  }
601 
602  // output data
603  id = 0ull;
604  for (const auto& outputDataSlot : node.second->outputDataConnections) {
605  // this connection type can make multiple connections, so two for loops are needed
606  for (const auto& connection : outputDataSlot) {
607  auto& remoteConn = connection.first->inputDataConnections[connection.second];
608  if (remoteConn.first != node.second.get() || remoteConn.second != id) {
609  res.addEntry("EUKN", "Data connection doesn't connect back",
610  {{"Left Node", node.second->id()}, {"Right Node", connection.first->id()},
611  {"Right input ID", connection.second}});
612  }
613  }
614  ++id;
615  }
616 
617  // input exec
618  id = 0ull;
619  for (const auto& inputExecSlot : node.second->inputExecConnections) {
620  // this connection type can make multiple connections, so two for loops are needed
621  for (const auto& connection : inputExecSlot) {
622  auto& remoteConn = connection.first->outputExecConnections[connection.second];
623  if (remoteConn.first != node.second.get() || remoteConn.second != id) {
624  res.addEntry("EUKN", "Exec connection doesn't connect back",
625  {{"Left Node", connection.first->id()}, {"Right Node", node.second->id()},
626  {"Left output ID", connection.second}}); // TODO: better diagnostics
627  }
628  }
629  ++id;
630  }
631 
632  // output exec
633  id = 0ull;
634  for (const auto& connection : node.second->outputExecConnections) {
635  bool connectsBack = false;
636  for (const auto& remoteConnection :
637  connection.first->inputExecConnections[connection.second]) {
638  if (remoteConnection.second == id && remoteConnection.first == node.second.get()) {
639  connectsBack = true;
640  break;
641  }
642  }
643 
644  if (!connectsBack) {
645  res.addEntry("EUKN", "Exec connection doesn't connect back",
646  {{"Left Node", node.second->id()}, {"Right Node", connection.first->id()},
647  {"Left output ID", id}});
648  }
649 
650  ++id;
651  }
652  }
653 
654  // make sure all entires have the right type
655 
656  return res;
657 }
