34  auto matching = nodesWithType("lang", "entry");
35 
36  if (matching.size() == 1) {
37  auto& vec = matching[0]->type().dataOutputs();
38  // make sure it has the same signature as the method

40  return nullptr;

96 Result GraphFunction::removeNode(NodeInstance* nodeToRemove) {
97  Result res;
98 
99  // disconnect it's connections
100 
101  // disconnect input exec
102  for (const auto& execSlot : nodeToRemove->inputExecConnections) {
103  for (const auto& pair : execSlot) {
104  if (pair.first != nullptr) { res += disconnectExec(*pair.first, pair.second); }
105  }
106  }
107  // disconnect output exec
108  auto ID = 0ull;
109  for (const auto& pair : nodeToRemove->outputExecConnections) {
110  if (pair.first != nullptr) { res += disconnectExec(*nodeToRemove, ID); }
111  ++ID;
112  }
113 
114  // disconnect input data
115  for (const auto& pair : nodeToRemove->inputDataConnections) {
116  if (pair.first != nullptr) {
117  res += disconnectData(*pair.first, pair.second, *nodeToRemove);
118  }
119  }
120 
121  // disconnect output data
122  ID = 0ull;
123  for (const auto& dataSlot : nodeToRemove->outputDataConnections) {
124  for (const auto& pair : dataSlot) {
125  if (pair.first != nullptr) { disconnectData(*nodeToRemove, ID, *pair.first); }
126  }
127  ++ID;
128  }
129  // then delete the node
130  nodes().erase(nodeToRemove->id());
131 
132  return res;
133 }
