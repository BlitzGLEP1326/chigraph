91 JsonModule::JsonModule(Context& cont, std::string fullName, const nlohmann::json& json_data,
92  Result* res)
93  : ChigModule(cont, fullName) {
94  Expects(res != nullptr);
95 
96  // load dependencies
97  {
98  auto iter = json_data.find("dependencies");
99  if (iter == json_data.end()) {
100  res->addEntry("E38", "No dependencies element in module", {});
101  return;
102  }
103  if (!iter->is_array()) {
104  res->addEntry("E39", "dependencies element isn't an array", {});
105  return;
106  }
107 
108  for (const auto& dep : *iter) {
109  if (!dep.is_string()) {
110  res->addEntry("E40", "dependency isn't a string", {{"Actual Data", dep}});
111  continue;
112  }
113  *res += addDependency(dep);
114 
115  if (!*res) { return; }
116  }
117  }
118 
119  // load graphs
120  {
121  auto iter = json_data.find("graphs");
122  if (iter == json_data.end()) {
123  res->addEntry("E41", "no graphs element in module", {});
124  return;
125  }
126  if (!iter->is_array()) {
127  res->addEntry("E42", "graph element isn't an array", {{"Actual Data", *iter}});
128  return;
129  }
130  mFunctions.reserve(iter->size());
131  for (const auto& graph : *iter) {
132  auto newf = std::make_unique<GraphFunction>(*this, graph, *res);
133 
134  if (!*res) { return; }
135 
136  Expects(newf != nullptr);
137 
138  mFunctions.push_back(std::move(newf));
139  }
140  }
141 }
